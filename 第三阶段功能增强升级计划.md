```javascript
// 找到 upload/from-url 接口，添加探测逻辑
app.post('/api/video2/from-url', upload.none(), async (req, res) => {
  const { url, projectId } = req.body;
  
  if (!url) {
    return res.status(400).json({ error: 'URL 不能为空' });
  }
  
  // 1. 验证 URL 格式
  try {
    new URL(url);
  } catch {
    return res.status(400).json({ error: 'URL 格式无效' });
  }
  
  // 2. HEAD 请求探测可访问性（5秒超时）
  try {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 5000);
    
    const probe = await fetch(url, {
      method: 'HEAD',
      signal: controller.signal
    });
    clearTimeout(timeout);
    
    if (!probe.ok) {
      return res.status(400).json({
        error: `URL 无法访问：HTTP ${probe.status} ${probe.statusText}`
      });
    }
    
    // 3. 验证 Content-Type
    const contentType = probe.headers.get('content-type') || '';
    const validTypes = ['image/', 'video/'];
    if (!validTypes.some(t => contentType.startsWith(t))) {
      return res.status(400).json({
        error: `URL 不是有效的图片或视频资源（Content-Type: ${contentType || '未知'}）`
      });
    }
  } catch (err) {
    if (err.name === 'AbortError') {
      return res.status(400).json({ error: 'URL 探测超时，请检查网络连接' });
    }
    return res.status(400).json({ error: `URL 探测失败：${err.message}` });
  }
  
  // 4. 继续原有的下载和上传逻辑...
});
```

```javascript
// 1. 在 video2Projects 对象中添加场次统计方法
getSceneStats: async (projectId) => {
  const rows = await new Promise((resolve, reject) => {
    video2Db.all(`
      SELECT 
        v.sceneId,
        s.name as sceneName,
        v.status,
        COUNT(*) as cnt
      FROM videos v
      LEFT JOIN video2_scenes s ON v.sceneId = s.id
      WHERE v.projectId = ? AND v.deleted = 0 AND v.status != 'trash'
      GROUP BY v.sceneId, v.status
    `, [projectId], (err, rows) => {
      if (err) reject(err);
      else resolve(rows);
    });
  });
  
  // 整理成场次维度
  const sceneMap = {};
  rows.forEach(r => {
    if (!sceneMap[r.sceneId]) {
      sceneMap[r.sceneId] = {
        id: r.sceneId,
        name: r.sceneName || '未分类',
        pending: 0,
        done: 0,
        total: 0
      };
    }
    if (r.status === 'done') {
      sceneMap[r.sceneId].done = r.cnt;
    } else {
      sceneMap[r.sceneId].pending += r.cnt;
    }
    sceneMap[r.sceneId].total += r.cnt;
  });
  
  return Object.values(sceneMap);
}
```

**技术细节**：

#### 后端修改

**文件**：`server/index.js`

```javascript
const path = require('path');
const fs = require('fs');
const DB_PATH = path.join(__dirname, 'video2.db');
const BACKUP_DIR = path.join(__dirname, 'backups');
const KEEP_DAYS = 7;

// 确保备份目录存在
if (!fs.existsSync(BACKUP_DIR)) {
  fs.mkdirSync(BACKUP_DIR, { recursive: true });
}

// 1. 创建备份函数
function createBackup() {
  const timestamp = new Date().toISOString().slice(0, 10);
  const backupPath = path.join(BACKUP_DIR, `video2_${timestamp}_${Date.now()}.db`);
  
  try {
    fs.copyFileSync(DB_PATH, backupPath);
    console.log(`[backup] 备份已创建: ${backupPath}`);
    
    // 清理旧备份
    cleanupOldBackups();
    return { success: true, path: backupPath };
  } catch (err) {
    console.error('[backup] 备份失败:', err);
    return { success: false, error: err.message };
  }
}

// 2. 清理旧备份
function cleanupOldBackups() {
  const files = fs.readdirSync(BACKUP_DIR).filter(f => f.endsWith('.db'));
  const now = Date.now();
  
  files.forEach(f => {
    const filePath = path.join(BACKUP_DIR, f);
    const stats = fs.statSync(filePath);
    const days = (now - stats.mtimeMs) / 86400000;
    
    if (days > KEEP_DAYS) {
      fs.unlinkSync(filePath);
      console.log(`[backup] 已删除过期备份: ${f}`);
    }
  });
}

// 3. 获取备份列表
function getBackupList() {
  if (!fs.existsSync(BACKUP_DIR)) return [];
  
  const files = fs.readdirSync(BACKUP_DIR)
    .filter(f => f.startsWith('video2_') && f.endsWith('.db'))
    .map(f => {
      const filePath = path.join(BACKUP_DIR, f);
      const stats = fs.statSync(filePath);
      return {
        filename: f,
        path: filePath,
        size: stats.size,
        createdAt: stats.mtime.toISOString()
      };
    })
    .sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt));
  
  return files;
}

// 4. 恢复备份
function restoreBackup(filename) {
  const backupPath = path.join(BACKUP_DIR, filename);
  if (!fs.existsSync(backupPath)) {
    return { success: false, error: '备份文件不存在' };
  }
  
  // 先备份当前数据库
  const tempPath = path.join(BACKUP_DIR, `video2_before_restore_${Date.now()}.db`);
  fs.copyFileSync(DB_PATH, tempPath);
  
  try {
    fs.copyFileSync(backupPath, DB_PATH);
    console.log(`[backup] 已恢复备份: ${filename}`);
    return { success: true };
  } catch (err) {
    // 恢复失败，恢复临时备份
    fs.copyFileSync(tempPath, DB_PATH);
    return { success: false, error: err.message };
  } finally {
    // 删除临时备份
    fs.unlinkSync(tempPath);
  }
}

// 5. 添加定时任务（每天凌晨 3 点）
setInterval(() => {
  const now = new Date();
  if (now.getHours() === 3 && now.getMinutes() === 0) {
    createBackup();
  }
}, 60000); // 每分钟检查一次

// 6. 添加备份接口
app.get('/api/video2/backup/list', (req, res) => {
  const list = getBackupList();
  res.json({ success: true, data: list });
});

app.post('/api/video2/backup/create', (req, res) => {
  const result = createBackup();
  res.json(result);
});

app.post('/api/video2/backup/restore', express.json(), (req, res) => {
  const { filename } = req.body;
  if (!filename) {
    return res.status(400).json({ error: '缺少 filename 参数' });
  }
  const result = restoreBackup(filename);
  if (result.success) {
    res.json({ success: true, message: '恢复成功，请刷新页面' });
  } else {
    res.status(500).json({ error: result.error });
  }
});
```

```javascript
const ADMIN_TOKEN = process.env.ADMIN_TOKEN || '';

// 1. 鉴权中间件
function adminAuth(req, res, next) {
  // 未配置 Token 时跳过鉴权（开发模式）
  if (!ADMIN_TOKEN) return next();
  
  const token = req.headers['x-admin-token'];
  if (token === ADMIN_TOKEN) {
    next();
  } else {
    res.status(401).json({ error: 'Unauthorized: Invalid or missing token' });
  }
}
```

```javascript
app.post('/api/video2/video/extract-frames', upload.single('video'), async (req, res) => {
  const { timestamps } = req.body;  // JSON array of timestamps in seconds
  const videoPath = req.file.path;
  
  if (!timestamps || !Array.isArray(JSON.parse(timestamps))) {
    return res.status(400).json({ error: '缺少 timestamps 参数' });
  }
  
  const times = JSON.parse(timestamps);
  const frames = [];
  
  for (let i = 0; i < times.length; i++) {
    const outputPath = path.join(OSS_TEMP_DIR, `frame_${Date.now()}_${i}.jpg`);
    
    await new Promise((resolve, reject) => {
      exec(`ffmpeg -ss ${times[i]} -i "${videoPath}" -vframes 1 -q:v 2 "${outputPath}"`, (err) => {
        if (err) reject(err);
        else resolve();
      });
    });
    
    // 上传到 OSS
    const ossKey = `temp/frames/${path.basename(outputPath)}`;
    await ossClient.put(ossKey, outputPath);
    
    frames.push({
      timestamp: times[i],
      url: getOssUrl(ossKey)
    });
    
    // 清理临时文件
    fs.unlinkSync(outputPath);
  }
  
  // 清理原视频
  fs.unlinkSync(videoPath);
  
  res.json({ success: true, data: frames });
});
```

```javascript
const sseClients = new Map();

// SSE 端点
app.get('/api/video2/ai/task/:taskId/stream', (req, res) => {
  const { taskId } = req.params;
  
  res.writeHead(200, {
    'Content-Type': 'text/event-stream',
    'Cache-Control': 'no-cache',
    'Connection': 'keep-alive'
  });
  
  // 保存连接
  const clients = sseClients.get(taskId) || [];
  clients.push(res);
  sseClients.set(taskId, clients);
  
  req.on('close', () => {
    const updated = (sseClients.get(taskId) || []).filter(c => c !== res);
    if (updated.length > 0) {
      sseClients.set(taskId, updated);
    } else {
      sseClients.delete(taskId);
    }
  });
});

// 推送函数
function notifyTaskUpdate(taskId, data) {
  const clients = sseClients.get(taskId) || [];
  clients.forEach(res => {
    res.write(`data: ${JSON.stringify(data)}\n\n`);
  });
}
```

## 6. 风险与注意事项

### 安全风险

| 风险项 | 影响 | 应对措施 |
|--------|------|---------|
| 备份文件泄露 | 数据库内容暴露 | 备份目录不暴露在 Web 根目录；定期清理旧备份 |
| API Token 泄露 | 设置接口被未授权修改 | Token 存环境变量；前端不硬编码 |
| 恢复备份出错 | 数据丢失 | 恢复前自动备份当前版本；二次确认 |

### 性能风险

| 风险项 | 影响 | 应对措施 |
|--------|------|---------|
| PDF 导出大项目超时 | 导出失败 | 大项目使用异步任务；添加进度提示 |
| 视频抽帧占用资源 | 服务器卡顿 | 限制并发数；超时自动取消 |
| SSE 连接过多 | 内存占用高 | 限制单用户连接数；超时自动断开 |

### 兼容性风险

| 风险项 | 影响 | 应对措施 |
|--------|------|---------|
| PDF/Excel 库版本冲突 | 编译失败 | 使用与现有依赖兼容的版本 |
| SSE 旧浏览器不支持 | 功能不可用 | 检测 EventSource 支持，不支持时降级为轮询 |
